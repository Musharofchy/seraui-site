## Basic Usage

```tsx
import GlitchVault from "./glitchvault";

export default function MyComponent() {
  return (
    <GlitchVault
      className="w-full max-w-sm border rounded-lg"
      glitchColor="#0AF0F0"
      glitchRadius={120}
    >
      <div className="p-6">
        <h2>Your Content Here</h2>
        <p>Any content can go inside the GlitchVault</p>
      </div>
    </GlitchVault>
  );
}
```



## Props

<MdxTable>
| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Content to display inside the component |
| `className` | `string` | - | Additional CSS classes for styling |
| `glitchColor` | `string` | `"#0AF0F0"` | Hex color for the glitch effect |
| `glitchRadius` | `number` | `120` | Radius of the mouse interaction area |
| `performanceMode` | `"high" \| "balanced" \| "low"` | `"balanced"` | Performance optimization level |
| `disabled` | `boolean` | `false` | Disable the glitch effect |
| `theme` | `"light" \| "dark" \| "auto"` | - | Theme variant (future feature) |
</MdxTable>

## Examples

### Different Colors
```tsx
// Cyan glitch effect (default)
<GlitchVault glitchColor="#0AF0F0">
  <div className="p-6">Cyan Effect</div>
</GlitchVault>

// Red glitch effect
<GlitchVault glitchColor="#FF0040">
  <div className="p-6">Red Effect</div>
</GlitchVault>

// Green glitch effect
<GlitchVault glitchColor="#00FF41">
  <div className="p-6">Green Effect</div>
</GlitchVault>
```

### Performance Modes
```tsx
// High performance mode (more responsive)
<GlitchVault performanceMode="high">
  <div className="p-6">High Performance</div>
</GlitchVault>

// Low performance mode (better for mobile)
<GlitchVault performanceMode="low">
  <div className="p-6">Low Performance</div>
</GlitchVault>
```

### Disabled State
```tsx
// Disable the glitch effect
<GlitchVault disabled>
  <div className="p-6">No Glitch Effect</div>
</GlitchVault>
```
