## Usage
```tsx
import SparklesText from './sparklestext';

function App() {
  return (
    <div className="min-h-screen bg-gray-50 dark:bg-gray-900 p-8">
      <SparklesText
        as="h1"
        className="text-5xl md:text-7xl font-bold text-white tracking-tight"
        sparkleCount={15}
        sparkleSize={18}
        colors={{ first: '#fde047', second: '#f97316' }}
      >
        Starship
      </SparklesText>
    </div>
  );
}
```
